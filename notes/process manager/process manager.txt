//memasang pm2 pada ssh
npm install -g pm2

//Setelah proses install selesai, masuk ke directory notes-app-back-end dengan perintah cd notes-app-back-end:
pm2 start npm --name "notes-api" -- run "start-prod" 

//Di pm2, kita dapat me-restart proses secara manual dengan cara
pm2 restart notes-api

//Atau, dapat menghentikan prosesnya dengan cara
pm2 stop notes-api
